        }
    }, time);

function fetchAll(tasks, success, error) {
    var results = Array(tasks.length);
    var count = 0;
    for (var i = 0; i < tasks.length; i++) {
        (function(i) {
            tasks[i](function(resolve, data) {
                if (resolve) {
                    count++;
                    results[i] = data;
                    if (count === tasks.length) {
                        success(results);
                    }
                } else {
                    error(data);
                }
            })
        })(i);
    }
}
