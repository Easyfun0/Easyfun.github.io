JavaScript-ES6語法介紹

Posted on 2019-09-11

ES6語法介紹-變數範圍鍊(Variable Scoping)

let和var不同在於變數有效範圍界定的差別。
🦊var是以function界定變數的有效範圍。
🦊let是以區塊界定有效範圍，例:if{…}(…),for(…){…}等等

範例6-1

ES5

(function() {
    var a = 5;
    (function() {
        var a = 6;
        console.log(a);
    })();
    console.log(a);
})();

ES6

{
    let a = 5;
    if(false){
        let a = 6;
        console.log(a);
    }
    console.log(a);
}
let在for迴圈中
for (let i = 0; i < 1000; i++){
    setTimeout(function () {
        console.log(i);
    }, i * 1000);
}

let看起來好像沒有hoisting

{
    console.log(a);
    let a = 0;
}

let其實有hoisting但無法操作(又稱死區Dead Zone)
初始化之前不能使用變數

{
    let a = 2;
    {
        console.log(a);
        let a = 1;
    }
}

ES6語法介紹-常數(Constants)

JavaScript並沒有常數的概念，大多數都是依照命名習慣，以全大寫表示一個常數。從ES6開始加入const定義常數。
🐥 不可重新指定值,不准許重複宣告,不准許宣告前使用

範例6-2

ES5

Object.defineProperty(typeof global === "object" ? global : window, "PI1", {
    value: 3.1415926,
    enumerable: true,
    writable: false,
    configurable: false
});
console.log(PI1); //3.1415926
PI1 = 100;
console.log(PI1); //3.1415926

ES6

const PI2 = 3.14;
console.log(PI2); //3.1415926
PI2 = 100; //Assignment to constant variable.
console.log(PI2); //3.1415926

範例6-3

ES5

Object.defineProperty(typeof global === "object" ? global : window, "DATA1", {
    value: {
        content: "abc"
    },
    enumerable: true,
    writable: false,
    configurable: false
});
DATA1.content = "xyz";
console.log(DATA1.content);

ES6
🐥不可重新指定值，但物件可修改屬性內容

const DATA2 = {
    content: "abc"
}


DATA2.content = "xyz";
console.log(DATA2.content);

ES6語法介紹-函式範圍鏈(Function Scoping)

在區塊內的function如同let一樣

範例6-4

ES5

(function() {
    function foo() { return 1; }
    (function() {
        function foo() { return 2; }
        console.log(foo());  //2
    })();
    console.log(foo());  //1
})();

ES6

{
    function foo() { return 1;}
    {
        function foo() { return 2;}
        console.log(foo());
    }
    console.log(foo());
}

🐣ES6採用Block Level Function Declarations 界定Function範圍，由於IE8~IE10不支援ES6，仍然採用hoisting界定function範圍

在IE8會印出a
if (true) {
    function foo() {
        return "a";
    }
} else {
    function foo() {
        return "b";
    }
}
console.log(foo());

ES6語法介紹-箭頭函式(Arrow Functions)

為了簡化函式，箭頭函式語法更短，以及不改變this的特性，所有的箭頭函式都必須是匿名函式

範例6-5

ES5

var input = [10, 20, 30, 40, 50];
var output = input.map(function(value) {
    return value+1;
});
console.log(output);

ES6

var input = [10, 20, 30, 40, 50];
var output = input.map(value => {
    return value + 1;
});
console.log(output);

範例6-6

箭頭函式允許的表示方式:

1.沒有參數時要有小括號
2.只有一個參數時可以省略小括號
3.兩個以上參數要有小括號
4.只有一行回傳值可省略大括號

ES5
function() {
}
ES6
() => {
}
ES5
function(value) {
}
ES6
value => {
}
ES5
function(value, index) {
}
ES6
(value, index) => {
}
ES5
function(a) {
    return a*a;
}
ES6
a => a*a

範例6-7

箭頭函式中的this(arguments)不會改變指向的參考

ES5

document.show1 = function() {
    var that = this;
    ["how", "are", "you"].forEach(function (value) {
        console.log(this);
        var h2 = document.createElement("h2");
        h2.innerText = value;
        that.body.appendChild(h2);
    });
};
document.show1();

ES6

document.show2 = function() {     //this在document而不是window
    ["how", "are", "you"].forEach(value => {
        var h2 = document.createElement("h2");
        h2.innerText = value;
        this.body.appendChild(h2);
    });

};
document.show2();

ES6語法介紹-函式參數擴展(Extented Parameter Handling)

以往不能直接為函式參數指定預設值，只能採用其他變通方法。
ES6支援函式參數具備預設值的功能，讓程式碼更簡潔

範例6-8

ES5

function link1(height, color, url) {
    var height = height !== undefined ? height : 50;
    var color = color !== undefined ? color : "red";
    var url = url !== undefined ? url : "http://www.abc.com";
    console.log(height, color, url);
}
link1(null, "Blue");
link1(undefined, "", "http://www.xyz.com");

ES6

function link2(height = 50, color = "red", url = "www.google.com") {
    console.log(height, color, url);
}
link2(undefined, "Blue");
link2(undefined, "", "http://www.xyz.com");

範例6-9

arguments取得所有傳入參數。
…則是取得剩餘的參數。
ES5

function run1(a, b) {
    console.log(arguments);
}
run1(10, 20, 30);

ES6

function run2(a, ...params) {
    console.log(a, params);
}
run2(10, 20, 30);

範例6-10

也可透過…合併陣列

ES5

var xyz = ["x", "y", "z"];
var abc = ["a", "b", "c"];
var out = abc.concat(xyz);
console.log(out);

ES6

var xyz = ["x", "y", "z"];
var abc = ["a", "b", "c"];
var out = [...abc, ...xyz];
console.log(out);

也可用在類別的參數上

function Run(...params) {
    console.log(params);
}
var args = [10, 20, 30];

ES5

var run = new (Function.prototype.bind.apply(Run, [null].concat(args)));
//bind產生新函式

ES6

var run = new Run(...args);
