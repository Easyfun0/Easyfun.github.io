JavaScript-ES6語法介紹

Posted on 2019-09-11

ES6語法介紹-變數範圍鍊(Variable Scoping)

let和var不同在於變數有效範圍界定的差別。
🦊var是以function界定變數的有效範圍。
🦊let是以區塊界定有效範圍，例:if{…}(…),for(…){…}等等

範例6-1

ES5

(function() {
    var a = 5;
    (function() {
        var a = 6;
        console.log(a);
    })();
    console.log(a);
})();

ES6

{
    let a = 5;
    if(false){
        let a = 6;
        console.log(a);
    }
    console.log(a);
}
let在for迴圈中
for (let i = 0; i < 1000; i++){
    setTimeout(function () {
        console.log(i);
    }, i * 1000);
}

let看起來好像沒有hoisting

{
    console.log(a);
    let a = 0;
}

let其實有hoisting但無法操作(又稱死區Dead Zone)
初始化之前不能使用變數

{
    let a = 2;
    {
        console.log(a);
        let a = 1;
    }
}

ES6語法介紹-常數(Constants)

JavaScript並沒有常數的概念，大多數都是依照命名習慣，以全大寫表示一個常數。從ES6開始加入const定義常數。
🐥 不可重新指定值,不准許重複宣告,不准許宣告前使用

範例6-2

ES5

Object.defineProperty(typeof global === "object" ? global : window, "PI1", {
    value: 3.1415926,
    enumerable: true,
    writable: false,
    configurable: false
});
console.log(PI1); //3.1415926
PI1 = 100;
console.log(PI1); //3.1415926

ES6

const PI2 = 3.14;
console.log(PI2); //3.1415926
PI2 = 100; //Assignment to constant variable.
console.log(PI2); //3.1415926

範例6-3

ES5

Object.defineProperty(typeof global === "object" ? global : window, "DATA1", {
    value: {
        content: "abc"
    },
    enumerable: true,
    writable: false,
    configurable: false
});
DATA1.content = "xyz";
console.log(DATA1.content);

ES6
🐥不可重新指定值，但物件可修改屬性內容

const DATA2 = {
    content: "abc"
}


DATA2.content = "xyz";
console.log(DATA2.content);

ES6語法介紹-函式範圍鏈(Function Scoping)

在區塊內的function如同let一樣

範例6-4

ES5

(function() {
    function foo() { return 1; }
    (function() {
        function foo() { return 2; }
        console.log(foo());  //2
    })();
    console.log(foo());  //1
})();

ES6

{
    function foo() { return 1;}
    {
        function foo() { return 2;}
        console.log(foo());
    }
    console.log(foo());
}

🐣ES6採用Block Level Function Declarations 界定Function範圍，由於IE8~IE10不支援ES6，仍然採用hoisting界定function範圍

在IE8會印出a
if (true) {
    function foo() {
        return "a";
    }
} else {
    function foo() {
        return "b";
    }
}
console.log(foo());

ES6語法介紹-箭頭函式(Arrow Functions)

為了簡化函式，箭頭函式語法更短，以及不改變this的特性，所有的箭頭函式都必須是匿名函式

範例6-5

ES5

var input = [10, 20, 30, 40, 50];
var output = input.map(function(value) {
    return value+1;
});
console.log(output);

ES6

var input = [10, 20, 30, 40, 50];
var output = input.map(value => {
    return value + 1;
});
console.log(output);

範例6-6

箭頭函式允許的表示方式:

1.沒有參數時要有小括號
2.只有一個參數時可以省略小括號
3.兩個以上參數要有小括號
4.只有一行回傳值可省略大括號

ES5
function() {
}
ES6
() => {
}
ES5
function(value) {
}
ES6
value => {
}
ES5
function(value, index) {
}
ES6
(value, index) => {
}
ES5
function(a) {
    return a*a;
}
ES6
a => a*a

ES6語法介紹-函式參數擴展(Extented Parameter Handling)

以往不能直接為函式參數指定預設值，只能採用其他變通方法。
ES6支援函式參數具備預設值的功能，讓程式碼更簡潔

範例6-8

ES5

function link1(height, color, url) {
    var height = height !== undefined ? height : 50;
    var color = color !== undefined ? color : "red";
    var url = url !== undefined ? url : "http://www.abc.com";
    console.log(height, color, url);
}
link1(null, "Blue");
link1(undefined, "", "http://www.xyz.com");

ES6

function link2(height = 50, color = "red", url = "www.google.com") {
    console.log(height, color, url);
}
link2(undefined, "Blue");
link2(undefined, "", "http://www.xyz.com");

範例6-9

arguments取得所有傳入參數。
…則是取得剩餘的參數。
ES5

    console.log(arguments);
}
run1(10, 20, 30);

ES6

function run2(a, ...params) {
    console.log(a, params);
}
run2(10, 20, 30);

範例6-10

也可透過…合併陣列

ES5

ES6

    console.log(params);

ES6

var run = new Run(...args);

ES6語法介紹-類別(Class)

ES6類別語法沒有改變JavaScript基於prototype-based本質，只是在提供一種語法糖，主要目的是可更貼近傳統物件導向語言寫法

範例6-23

ES5

function Shape1(id, x, y) {
    this.id = id;
    this.move(x, y);
};
Shape1.prototype.move = function(x, y) {
    this.x = x;
    this.y = y;
};
var a = new Shape1("a", 0, 0);
console.log(a);

ES6

class Shape2{
    constructor(id, x, y) {  //定義建構式
        this.id = id;
        this.move(x,y)
    }
    move(x, y) {        //不是constructor都是
        this.x = x;
        this.y = y;
    }
}

var a = new Shape2("a", 0, 0);
console.log(a);

範例6-24

ES6繼承更簡潔
ES5

Function.prototype.inherits = function(superCtor) {
    Object.setPrototypeOf(this.prototype, superCtor.prototype);
    this.super = superCtor.prototype;
};
function Shape1(id, x, y) {
    this.id = id;
    this.move(x, y);
};
Shape1.prototype.move = function(x, y) {
    this.x = x;
    this.y = y;
};
function Rectangle1(id, x, y, width, height) {
    Shape1.call(this, id, x, y);
    this.width = width;
    this.height = height;
};
Rectangle1.inherits(Shape1);
function Circle1(id, x, y, radius) {
    Shape1.call(this, id, x, y);
    this.radius = radius;
};
Circle1.inherits(Shape1);
var a = new Rectangle1("a", 0, 0, 100, 100);
console.log(a);
var b = new Circle1("b", 0, 0, 100);
console.log(b);

ES6
原型方法則是透過super.move(x,y)就能呼叫父類別的原型方法

class Shape2 {
    constructor(id, x, y) {
        this.id = id
        this.move(x, y)
    }
    move(x, y) {
        this.x = x
        this.y = y
    }
}
class Rectangle2 extends Shape2{
    constructor(id, x, y, width, height) {
        super(id, x, y);
        this.width = width;
        this.height = height;
    }
    move(x,y) {   //super當作父類別
        super.move(x,y);
        console.log("Rectangle2 go");
    }
}

ES6語法介紹-生成器(Generator)

生成器為可暫時跳離的函式，因為一般函式遇到return就會直接跳離，但Generator函式執行到yield時會暫停且跳離，並且將值回傳，直到再次呼叫Generator物件的next方法，才會繼續回到函式中，這個過程可透過for…of迴圈自動實作

範例6-27

ES5

function range1(start, end, step) {
    var list = [];
    while (start < end) {
        list.push(start);
        start += step;
    }
    return list;
}
var r = range1(0, 10, 2);
for (var i = 0; i < r.length; i++) {
    console.log(r[i]); // 0, 2, 4, 6, 8
}

ES6 內建迭代器

function* range2(start,end,step) {  //暫停
    while (start < end) {
        yield start;
        start += step;
    }
}
for (let i of range2(0, 10, 2)) {
    console.log(i);
};

ES6語法介紹-非同步處理(Promise&Async…Await)

會產生非同步的方法有setTimeout,setInterval,XMLHttpRequest,fetch,Promise…等等。
ES5處理非同步行為使用Callback會讓程式碼難以維護，ES6提供Promise是將非同步行為統一相同界面接口，透過then或catch去接收或捕捉非同步結果，不會產生多層的Callback結構，從ES7開始支援Async…Await語法，程式閱讀起來都是同步的，但實際上是用非同步處理

範例6-33

ES5

function asyncTask1(success, time, callback) {
    setTimeout(function() {
        if (success) {
            callback(true, time);
        } else {
            callback(false, time);
        }
    }, time);
}
//非同步同時完成
function fetchAll(tasks, success, error) {
    var results = Array(tasks.length);
    var count = 0;
    for (var i = 0; i < tasks.length; i++) {
        (function(i) {
            tasks[i](function(resolve, data) {
                if (resolve) {
                    count++;
                    results[i] = data;
                    if (count === tasks.length) {
                        success(results);
                    }
                } else {
                    error(data);
                }
            })
        })(i);
    }
}
fetchAll([
    function(cb) { asyncTask1(true, 1000, cb) },
    function(cb) { asyncTask1(true, 5000, cb) },
    function(cb) { asyncTask1(true, 2500, cb) }    
],function(data) {
    console.log("ok:", data);
},function(err) {
    console.log("err:", err);
});
//非同步依序完成
asyncTask1(true, 1000, function(resolve, data) {
    resolve && console.log("ok:", data);
    resolve && asyncTask1(true, 5000, function(resolve, data) {
        resolve && console.log("ok:", data);
        resolve && asyncTask1(true, 2500, function(resolve, data) {
            resolve && console.log("ok:", data);
        });
    });
});

ES6

function asyncTask2(success, time) {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            if (success) {
                resolve(time);
            } else {
                reject(time);
            }
        }, time);
    });
}
//非同步同時完成
Promise.all([
    asyncTask2(true, 1000),
    asyncTask2(true, 5000),
